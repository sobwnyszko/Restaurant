CREATE TABLE Rezerwacja (ID int(10) NOT NULL AUTO_INCREMENT, Data date NOT NULL, Godzina time(7) NOT NULL, `StolikNumer stolika` int(10) NOT NULL, ImięID int(10) NOT NULL, Nazwisko varchar(127) NOT NULL, `nr kontaktowy` char(11) NOT NULL, PRIMARY KEY (ID));
